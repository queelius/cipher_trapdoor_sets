\newglossaryentry{gls-preimageres}
{
    name={pre-image resistant},
    description={Given a hash function and a hash, it is difficult to find an
    input for the given hash function that outputs the given hash.}
}

\newglossaryentry{gls-fprate}
{
    name={false positive rate},
    description={Given an objective set $\Set{A}$ and a random approximate set
    $\ASet{A}(\fprate,\fnrate)$, a an element $x$ that is not a member of
    $\Set{A}$ is a member of $\ASet{A}$ with a probability $\fprate$ denoted the
    \emph{false positive rate}}.
}

\newglossaryentry{gls-fnrate}
{
    name={false negative rate},
    description={Given an objective set $\Set{B}$ and a random approximate set
    $\ASet{B}(\fprate,\fnrate)$, an element $x$ that is not a member of
    $\Set{B}$ is a member of $\ASet{B}(\fprate,\fnrate)$ with probability
    $\fnrate$ denoted the \emph{false negative rate}}.
}

\newglossaryentry{gls-preimageattack}
{
    name={pre-image attack},
    description={Given a hash function and a hash, searching for an input for
    the given hash function that outputs the given hash.}
}

\newglossaryentry{gls-randomoracle}
{
    name={random oracle},
    description={A theoretical function that maps every unique input from its
    domain to a uniform and random output in its range.}
}

\newglossaryentry{gls-approxset}
{
    name={random approximate set},
    description={A set $\ASet{A}(\fprate,\fnrate)$ is a \emph{random approximate
    set} of an objective set $\Set{A}$ if an element not in $\Set{A}$ is in
    $\ASet{A}(\fprate,\fnrate)$ with a probability $\fprate$ and an element in
    $\Set{A}$ is not in $\ASet{A}(\fprate,\fnrate)$ with a
    probability $\fnrate$.
}


\newglossaryentry{gls-posapproxset}
{
    name={random positive approximate set},
    description={A set $\PASet{A}(\fprate)$ is a \emph{random positive
    approximate set} of an objective set $\Set{A}$ if
    $\PASet{A} \supset \Set{A}$ and an element not in $\Set{A}$ is in
    $\PASet{A}(\fprate)$ with a probability denoted the \gls*{gls-fprate}.}
}

\newglossaryentry{gls-shs}
{
    name={Singular Hash Set},
    description={A theoretically \emph{optimal} implementation of the abstract
    data type of the \emph{positive oblivious set}.}
}

\newglossaryentry{gls-shmap}
{
    name={Singular Hash Map},
    description={A theoretically \emph{optimal} implementation of the abstract
    data type of the \emph{random approximate map}.}
}

\newglossaryentry{gls-pf}
{
    name={perfect hash function},
    description={A hash function that guarantees no collisions among members of
    a specified set.}
}

\newglossaryentry{gls-phf}
{
    name={Perfect Hash Filter},
    description={An implementation of the abstract data type of the \emph{random
    positive approximate set} that is a function of a \gls*{gls-pf}.}
}

\newglossaryentry{gls-blf}
{
    name={Bloom filter},
    description={A poular implementation of the abstract data type of the
    \emph{random positive approximate set}.}
}

\newglossaryentry{gls-nb}

    name={negative binomial distribution},

}

\newglossaryentry{gls-geo}
{
    name={geometric distribution},
    description={A distribution where we are interested in observing the number
    of Bernoulli trials needed to observe a a ``successes.''}
}

\newglossaryentry{gls-set}
